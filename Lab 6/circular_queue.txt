#include <iostream>
#include <string>
using namespace std;

class Node
{
public:
    string name;
    Node *next_ptr;

    Node(string name, Node *next_ptr = nullptr)
    { 
        this->name = name;
        this->next_ptr = next_ptr;
    }
};

class Queue
{
private:
    Node *front;
    Node *rear;

public:
    Queue(Node *front = nullptr)
    { 
        this->front = front;
        this->rear = front;
        if (front)
        {
            front->next_ptr = front; 
        }
    }

    void enqueue(Node *node)
    {
        if (!node)
            return; 

        node->next_ptr = nullptr; 

        if (!front)
        { 
            front = node;
            rear = node;
            node->next_ptr = front; 
        }
        else
        {
            rear->next_ptr = node;  
            rear = node;            
            node->next_ptr = front; 
        }
    }

    void display_queue()
    {
        if (!front)
        {
            cout << "Queue is empty" << endl;
            return;
        }

        Node *current = front;
        do
        {
            cout << current->name << " <- ";
            current = current->next_ptr;
        } while (current != front); 
        cout << "(circular)" << endl;
    }

    void dequeue()
    {
        if (!front)
        {
            cout << "Queue is empty" << endl;
            return;
        }

        Node *temp = front;
        if (front == rear)
        {
            front = nullptr;
            rear = nullptr;
        }
        else
        {
            front = front->next_ptr; 
            rear->next_ptr = front;
        }
        cout << "Removing: " << temp->name << endl;
        delete temp;
    }
};

int main()
{

    Node *node1 = new Node("Ali");
    Node *node2 = new Node("Ahmed");
    Node *node3 = new Node("Alee");
    Node *node4 = new Node("Abu");

    Queue queue(node1);
    queue.enqueue(node2);
    queue.enqueue(node3);
    queue.enqueue(node4);

    queue.display_queue();

    queue.dequeue();
    cout << "After Dequeue: ";
    queue.display_queue();

    return 0;
}

/* Expected output:
Ali <- Ahmed <- Alee <- Abu <- (circular)
Removing: Ali
After Dequeue: Ahmed <- Alee <- Abu <- (circular)
*/
